import React, { useState, useEffect, useRef, useCallback } from 'react';
import { useChat } from '@/hooks/useChat';
import { useScrollAnchor } from '@/hooks/useScrollAnchor';
import { Message, ChatSession, StreamingState, MessageRole } from '@/types/chat';
import { ScrollArea } from '@/components/ui/scroll-area';
import { MessageList } from './MessageList';
import { MessageInput } from './MessageInput';
import { MarkdownRenderer } from './MarkdownRenderer';
import { cn } from '@/lib/utils';

interface ChatInterfaceProps {
  sessionId: string;
  initialMessages?: Message[];
  onSessionUpdate?: (session: ChatSession) => void;
  className?: string;
}

export default function ChatInterface({
  sessionId,
  initialMessages = [],
  onSessionUpdate,
  className,
}: ChatInterfaceProps) {
  const scrollAreaRef = useRef<HTMLDivElement | null>(null);
  const bottomRef = useRef<HTMLDivElement | null>(null);
  const [streamingState, setStreamingState] = useState<StreamingState>('idle');
  const [pendingMessage, setPendingMessage] = useState<string>('');
  const [isComposing, setIsComposing] = useState(false);

  const { messages, sendMessage, isLoading, error, clearMessages } = useChat({
    sessionId,
    initialMessages,
    onUpdate: onSessionUpdate,
    streamingOptions: { enabled: true, chunkSize: 64, delayMs: 12 },
  });

  const scrollToBottom = useCallback((behavior: ScrollBehavior = 'smooth') => {
    bottomRef.current?.scrollIntoView({ behavior, block: 'end' });
  }, []);

  useEffect(() => {
    if (messages.length > 0 && streamingState !== 'streaming') {
      scrollToBottom();
    }
  }, [messages, streamingState, scrollToBottom]);

  const handleSubmit = useCallback(
    async (content: string, attachments?: File[]) => {
      if (!content.trim() || isLoading) return;
      setStreamingState('streaming');
      setPendingMessage('');
      try {
        await sendMessage({ role: 'user' as MessageRole, content: content, attachments: attachments });
      } catch (err) {
        console.error('[ChatInterface] sendMessage failed:', err);
        setStreamingState('error');
      } finally {
        setStreamingState('idle');
      }
    },
    [sendMessage, isLoading],
  );

  return (
    <div className={cn('flex h-full flex-col overflow-hidden bg-background', className)}>
      <ScrollArea ref={scrollAreaRef} className="flex-1 px-4 py-2">
        <MessageList
          messages={messages}
          streamingState={streamingState}
          renderMessage={(msg) => (
            <MarkdownRenderer
              key={msg.id}
              content={msg.content}
              role={msg.role}
              timestamp={msg.createdAt}
            />
          )}
        />
        <div ref={bottomRef} className="h-px w-full" aria-hidden="true" />
      </ScrollArea>
      <MessageInput
        value={pendingMessage}
        onChange={setPendingMessage}
        onSubmit={handleSubmit}
        disabled={isLoading}
        isComposing={isComposing}
        onCompositionStart={() => setIsComposing(true)}
        onCompositionEnd={() => setIsComposing(false)}
        placeholder="Ask anything… (⌘ Enter to send)"
      />
    </div>
  );
}
